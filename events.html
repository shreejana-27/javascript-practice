onClick -> when click in the html element
        ondblclick -> double click on the element
        onsubmit -> when a form is submitted
        onload -> when a page is loaded
        onchange -> when a input field value is changed
        onmouseover -> hover over a element
        onmouseout -> opposite of mouseover
        onkeyup -> when a key is pressed
        onkeydown -> same as onkeyup
        onmousemove -> when a mouse moves over an element
